




// question A
a. update the employee with employee_no 1002 job to 4, with effective 2020-01-01.


UPDATE employee_job_table
SET job_id = 4, effective_date = '2020-01-01'
WHERE employee_id = (
    SELECT employee_id
    FROM employee_profile_table
    WHERE employee_no = 1002
);


// question B
b. list out the employee name, employee no, effective date, job title.

SELECT 
    e.employee_name,
    e.employee_no,
    j.effective_date,
    p.job_name
FROM 
    employee_profile_table e
JOIN 
    employee_job_table j ON e.employee_id = j.employee_id
JOIN 
    job_profile_table p ON j.job_id = p.job_id
ORDER BY 
    e.employee_no;


// question C
c. list out the employee name, employee no, latest job title

    SELECT 
    e.employee_name,
    e.employee_no,
    p.job_name
FROM 
    employee_profile_table e
JOIN (
    SELECT 
        employee_id, 
        job_id,
        MAX(effective_date) as latest_effective_date
    FROM 
        employee_job_table
    GROUP BY 
        employee_id
) j ON e.employee_id = j.employee_id
JOIN 
    job_profile_table p ON j.job_id = p.job_id
ORDER BY 
    e.employee_no;


 // question D
d. delete all employee with employee_no 1000 job title, with effective date of 2020-01-01


DELETE FROM employee_job_table
WHERE employee_id = (
    SELECT employee_id
    FROM employee_profile_table
    WHERE employee_no = 1000
) AND effective_date = '2020-01-01';
